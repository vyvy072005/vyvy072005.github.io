n = int(input())       # количество чисел
s = 0       # начальное значение суммы
for i in range(1,n+1):       # цикл
    a = int(input())       # ввод числа последовательности
    if a < 50 and a > 10 :       # условие
        s += a       # прибавление числа к сумме
print(s)       # ввывод суммы

n = int(input())       # ввод первого числа последовательности
k = 0       # начальное значение количества
while n != 0:       # цикл
    if n % 6 == 0 and n % 10 == 0 :   # условие
        k += 1
    n = int(input())       # ввод числа последовательности
print(k)       # вывод количества
